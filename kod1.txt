program fibonacci;
var
  i, liczba: word;
  a, b, c: longint;
begin
  writeln('Podaj, ktora liczbe z ciagu Fibonacciego chcesz zobaczyc: ');
  readln(liczba);
  a := 1;
  b := 1;
  if liczba <= 2 then 
    writeln('Wynik: ', a)
  else 
  begin
    for i := 3 to liczba do 
    begin
      c := a+b;
      a := b;
      b := c;
    end;
    writeln('Wynik: ', c);
  end;
end.
